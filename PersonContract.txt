pragma solidity 0.8.30;

contract Person{
        string  name;
        uint age; 
        string location;

    function setPerson(
        string memory _name,
        uint _age,
        string memory _location
    ) public  {
        name = _name;
        age = _age;
        location = _location;
    }

    function getPerson() public  view returns (string memory,uint,string memory){
        return (name,age,location);
    }
}
